import useDocusaurusContext from '@docusaurus/useDocusaurusContext';

export const Value = ({name}) => {
  const {siteConfig} = useDocusaurusContext();
  const meta = Object.fromEntries(
    (siteConfig.themeConfig?.metadata || []).map(m => [m.name, m.content])
  );
  const map = { title: siteConfig.title, tagline: siteConfig.tagline, ...meta };
  return <>{map[name] || ''}</>;
};

export const TechBadges = () => {
  const {siteConfig} = useDocusaurusContext();
  const techs = (siteConfig.themeConfig?.metadata || [])
    .find(m => m.name === 'docusaurOpsProjectTechnologies')?.content || '';
  return (
    <>
      {techs.split(',').map((t, i) => t.trim()).filter(Boolean).map((t, i) => (
        <span key={i}>{i > 0 && ' · '}<code>{t}</code></span>
      ))}
    </>
  );
};
export const Link = ({name, label}) => {
  const {siteConfig} = useDocusaurusContext();
  const meta = Object.fromEntries(
    (siteConfig.themeConfig?.metadata || []).map(m => [m.name, m.content])
  );
  const url = meta[name] || '#';
  return <a href={url} target="_blank" rel="noopener noreferrer">{label || url}</a>;
};
