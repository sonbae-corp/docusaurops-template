export const Value = ({name}) => {
  const {siteConfig} = useDocusaurusContext();
  const meta = Object.fromEntries(
    (siteConfig.themeConfig?.metadata || []).map(m => [m.name, m.content])
  );
  const map = { title: siteConfig.title, tagline: siteConfig.tagline, ...meta };
  return <>{map[name] || ''}</>;
};

export const TechBadges = () => {
  const {siteConfig} = useDocusaurusContext();
  const techs = (siteConfig.themeConfig?.metadata || [])
    .find(m => m.name === 'docusaurOpsProjectTechnologies')?.content || '';
  const items = techs.split(',').map(t => t.trim()).filter(Boolean);
  if (!items.length) return null;
  return (
    <div className="flex flex-wrap gap-2 justify-center mt-2">
      {items.map((t, i) => (
        <span key={i} className="px-3 py-1 rounded-full text-sm font-mono font-semibold"
          style={{background:'var(--ifm-color-primary-lightest)', color:'var(--ifm-color-primary-darkest)'}}>
          {t}
        </span>
      ))}
    </div>
  );
};

export const MetaLink = ({name, label, fallback, className, style}) => {
  const {siteConfig} = useDocusaurusContext();
  const meta = Object.fromEntries(
    (siteConfig.themeConfig?.metadata || []).map(m => [m.name, m.content])
  );
  const url = meta[name];
  if (!url) return <span className="opacity-40 italic">{fallback || '—'}</span>;
  return <a href={url} target="_blank" rel="noopener noreferrer" className={className} style={style}>{label || url}</a>;
};

    <span className="w-1.5 h-1.5 rounded-full inline-block" style={{background: color}}></span>
    {label}
  </span>
);
